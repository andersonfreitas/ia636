Function iainterpollin

Synopse

Perform linear interpolation

  • v = iainterpollin(f, pts)
    • f: 1D, 2D or 3D image to be interpolated
    • pts: array DxN with the points to interpolate (one in each column)
    • v: array of the interpolated values

For each cell the following equation needs to be solved:

1D: v = a + bx

2D: v = a + bx + cy + exy

3D: v = a + bx + cy + dz + exy + fxz + gyz + hxyz

where x, y and z are the fractional part of the coordinate.

– 
  1 def iainterpollin(f, pts):
  2     # f - one, two or three dimention array
  3     # pts - array of points to interpolate:
  4     # throws IndexError if there are indices out of range in pts
  5     from iainterpollin import iainterpollin1D, iainterpollin2D, iainterpollin3D
  6     import numpy as np
  7     f = f.astype(float)
  8     if f.ndim == 1:
  9         return iainterpollin1D(f, np.ravel(pts))
 10     if f.ndim == 2:
 11         return iainterpollin2D(f, pts)
 12     if f.ndim == 3:
 13         return iainterpollin3D(f, pts)
 14 
 15 def iainterpollin1D(f, pts):
 16     # f - one dimention array
 17     # pts - array of points to interpolate:
 18     # throws IndexError if there are indices out of range in pts
 19     import numpy as np
 20 
 21     # integer indices
 22     ipts = np.floor(pts).astype(int)
 23 
 24     # fractional indices
 25     fpts = pts - ipts
 26 
 27     # workaround for the case some index equals the last valid index
 28     fpts[ipts>=f.shape[0]-1] += 1
 29     ipts[ipts>=f.shape[0]-1] -= 1
 30 
 31     I = ipts.copy()
 32     Ix = ipts.copy()+1
 33 
 34     a = f[I]
 35     b = f[Ix] - a
 36 
 37     return a + b*fpts

 40 def iainterpollin2D(f, pts):
 41     # f - bi-dimentional image
 42     # pts - array (2xN) of points to interpolate:
 43     # throws IndexError if there are indices out of range in pts
 44     # let's consider x the first and y the second dimention
 45     import numpy as np
 46 
 47     # integer indices
 48     ipts = np.floor(pts).astype(int)
 49 
 50     # fractional indices
 51     fpts = pts - ipts
 52 
 53     # workaround for the case some index equals the last valid index in the respective dimension
 54     fpts[0][ipts[0]>=f.shape[0]-1] += 1
 55     ipts[0][ipts[0]>=f.shape[0]-1] -= 1
 56     fpts[1][ipts[1]>=f.shape[1]-1] += 1
 57     ipts[1][ipts[1]>=f.shape[1]-1] -= 1
 58 
 59     I = ipts.copy()
 60     Ix = ipts.copy()
 61     Ix[0] += 1
 62     Iy = ipts.copy()
 63     Iy[1] += 1
 64     Ixy = Ix.copy()
 65     Ixy[1] += 1
 66 
 67     a = f[I[0], I[1]]
 68     b = f[Ix[0], Ix[1]] - a
 69     c = f[Iy[0], Iy[1]] - a
 70     e = f[Ixy[0], Ixy[1]] - a - b - c
 71 
 72     return a + b*fpts[0] + c*fpts[1] + e*fpts[0]*fpts[1]
 73 
 74 
 75 def iainterpollin3D(ff, pts):
 76     # ff - tri dimentional image
 77     # pts - array (3xN) of points to interpolate:
 78     # throws IndexError if there are indices out of range in pts
 79     # let's consider x the first dimension, y the second and z the third
 80     import numpy as np
 81 
 82     # integer indices
 83     ipts = np.floor(pts).astype(int)
 84 
 85     # fractional indices
 86     fpts = pts - ipts
 87 
 88     # workaround for the case some index equals the last valid index in the respective dimension
 89     fpts[0][ipts[0]>=ff.shape[0]-1] += 1
 90     ipts[0][ipts[0]>=ff.shape[0]-1] -= 1
 91     fpts[1][ipts[1]>=ff.shape[1]-1] += 1
 92     ipts[1][ipts[1]>=ff.shape[1]-1] -= 1
 93     fpts[2][ipts[2]>=ff.shape[2]-1] += 1
 94     ipts[2][ipts[2]>=ff.shape[2]-1] -= 1
 95 
 96     I = ipts.copy()
 97     Ix = ipts.copy()
 98     Ix[0] += 1
 99     Iy = ipts.copy()
100     Iy[1] += 1
101     Iz = ipts.copy()
102     Iz[2] += 1
103     Ixy = Ix.copy()
104     Ixy[1] += 1
105     Ixz = Ix.copy()
106     Ixz[2] += 1
107     Iyz = Iy.copy()
108     Iyz[2] += 1
109     Ixyz = Ixy.copy()
110     Ixyz[2] += 1
111 
112     a = ff[I[0], I[1], I[2]]
113     b = ff[Ix[0], Ix[1], Ix[2]] - a
114     c = ff[Iy[0], Iy[1], Iy[2]] - a
115     d = ff[Iz[0], Iz[1], Iz[2]] - a
116     e = ff[Ixy[0], Ixy[1], Ixy[2]] - a - b - c
117     f = ff[Ixz[0], Ixz[1], Ixz[2]] - a - b - d
118     g = ff[Iyz[0], Iyz[1], Iyz[2]] - a - c - d
119     h = ff[Ixyz[0], Ixyz[1], Ixyz[2]] - a - b - c - d - e - f - g
120 
121     return a + b*fpts[0] + c*fpts[1] + d*fpts[2] + e*fpts[0]*fpts[1] + f*fpts[0]*fpts[2] + g*fpts[1]*fpts[2] + h*fpts[0]*fpts[1]*fpts[2]
